function renderTabel() { if(!dataServerTerakhir) return; let tbody = document.getElementById('tabel-body'); tbody.innerHTML = ""; // Perbaikan: Gunakan Number() dan berikan nilai alternatif || 0 jika datanya rusak/NaN let totalMasuk = Number(dataServerTerakhir.total_masuk) || 0; let totalKeluar = Number(dataServerTerakhir.total_keluar) || 0; document.getElementById('txt-masuk').innerText = formatRupiah(totalMasuk); document.getElementById('txt-keluar').innerText = formatRupiah(totalKeluar); let totalKas = totalMasuk - totalKeluar; document.getElementById('txt-total-kas').innerText = formatRupiah(totalKas); let persen = totalMasuk > 0 ? Math.round((totalKas / totalMasuk) * 100) : 0; document.getElementById('txt-persen').innerText = persen + "%"; // Bagian riwayat transaksi if (dataServerTerakhir.riwayat && Array.isArray(dataServerTerakhir.riwayat)) { dataServerTerakhir.riwayat.forEach(row => { let tr = document.createElement('tr'); let badgeClass = row.jenis === "Masuk" ? "badge-jenis bg-masuk" : "badge-jenis bg-keluar"; let nominalRow = Number(row.jumlah) || 0; tr.innerHTML = `${row.keterangan || '-'} ${row.kategori || '-'} ${row.jenis || '-'} ${formatRupiah(nominalRow)}`; tbody.appendChild(tr); }); } }
